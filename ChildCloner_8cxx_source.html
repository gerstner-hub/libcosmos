59SubProc ChildCloner::run() {
60 if (m_executable.empty() || m_argv.empty()) {
61 cosmos_throw (UsageError(
62 "attempted to run a sub process w/o specifying an executable path and/or argv0"
63 ));
64 }
65
66 // use clone3() instead of fork():
67 //
68 // clone() allows us to get a pid fd for the child in a race free
69 // fashion
70 //
71 // clone3() has fork() semantics and is easier to use for this case
72 // than the older clone syscalls.
73 //
74 // NOTE: it turns out that clone3() is not yet supported in Valgrind,
75 // meaning we won't be able to run programs through Valgrind any more
76 // that employ this system call. clone2() is annoying to use because
77 // it doesn't have fork() semantics though ... maybe we can sit this
78 // out until Valgrind gets supports for clone3().
79 //
80 // Another alternative would be using regular fork() and create a
81 // PidFD from the child PID. As long as no one is collecting the child
82 // status via any of the wait() functions this would also be race
83 // free.
84 PidFD pidfd;
85 CloneArgs clone_args;
86 clone_args.setPidFD(pidfd);
87 clone_args.setFlags(CloneFlags{CloneSettings::CLEAR_SIGHAND, CloneSettings::PIDFD});
88
89 if (auto pid = proc::clone(clone_args); pid != std::nullopt) {
90 // parent process with child pid
91 return SubProc{*pid, pidfd};
92 }
93
94 // the child process -- let's do something!
95
96 try {
97 postFork();
98
99 auto argv = setup_argv(m_argv);
100
101 if (!m_env) {
102 proc::exec(argv[0], &argv);
103 } else {
104 auto envp = setup_env(m_env.value());
105 proc::exec(argv[0], &argv, &envp);
106 }
107 } catch (const CosmosError &ce) {
108 print_child_error("post fork/exec", ce.what());
109 // use something else than "1" which might help debugging this
110 // situation a bit.
111 proc::exit(ExitStatus{3});
112 }
113
114 // should never be reached
115 return SubProc();
116}

127void ChildCloner::postFork() {
128 if (m_post_fork_cb) {
129 m_post_fork_cb(*this);
130 }
131
132 if (m_sched_settings) {
133 try {
134 std::visit([](auto &&sched_settings) {
135 sched_settings.apply(ProcessID::SELF);
136 }, *m_sched_settings);
137 } catch(const std::exception &ex) {
138 // treat this as non-critical, the process can still
139 // run, even if not prioritized.
140 print_child_error("sched_setscheduler", ex.what());
141 }
142 }
143
144 resetSignals();
145
146 if (!m_cwd.empty()) {
147 fs::change_dir(m_cwd);
148 }
149
150 redirectFD(cosmos::stdout, m_stdout);
151 redirectFD(cosmos::stderr, m_stderr);
152 redirectFD(cosmos::stdin, m_stdin);
153
154 for (auto fd: m_inherit_fds) {
155 fd.setCloseOnExec(false);
156 }
157}
