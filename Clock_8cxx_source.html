53template <ClockType CLOCK>
54void Clock<CLOCK>::sleep(const TimeSpec<CLOCK> until) const {
55 while (true) {
56 auto res = clock_nanosleep(
57 to_integral(CLOCK),
58 TIMER_ABSTIME,
59 &until,
60 nullptr
61 );
62
63 const auto err = Errno{res};
64
65 switch(err) {
66 default: break;
67 case Errno::NO_ERROR: return;
68 case Errno::INTERRUPTED: {
69 if (auto_restart_syscalls) {
70 continue;
71 }
72 break;
73 }
74 }
75
76 cosmos_throw (ApiError(err));
77 }
