86 size_t nameLength() const {
87 // this actually involves padding, so look from a 8-byte
88 // offset from the end forwards, still faster for long strings.
89 auto len = m_entry->d_reclen - offsetof(struct dirent, d_name) - 1;
90
91 len = len < 8 ? 0 : len - 8;
92 auto endp = name() + len;
93
94 for (size_t i = 0; i < 8; i++) {
95 if (*(endp++) == '\0')
96 return endp - name() - 1;
