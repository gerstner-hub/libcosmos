87 template <typename SyncFunc>
88 void sync_helper(FileDescriptor &fd, SyncFunc sync_func, const char *errlabel) {
89 while (true) {
90 if (sync_func(to_integral(fd.raw())) == 0) {
91 return;
92 }
93 else {
94 switch(get_errno()) {
95 default: break;
96 case Errno::INTERRUPTED: {
97 if (auto_restart_syscalls)
98 continue;
99 }
100 }
101
102 cosmos_throw (ApiError(errlabel));
103 }
104 }
105 }
