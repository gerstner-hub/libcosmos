52std::string FileMode::symbolic() const {

55 if (canOwnerRead()) ret.push_back('r');

63 else ret.push_back('-');

65 if (canGroupRead()) ret.push_back('r');
66 else ret.push_back('-');

69 else ret.push_back('-');
70
71 if (isSetGID()) ret.push_back('s');
72 else if (canGroupExec()) ret.push_back('x');
73 else ret.push_back('-');
74
75 if (canOthersRead()) ret.push_back('r');
76 else ret.push_back('-');
77
78 if (canOthersWrite()) ret.push_back('w');

81 if (isSticky()) ret.push_back('t');
